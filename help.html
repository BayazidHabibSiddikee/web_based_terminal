blink.c — ATmega8/32/328P · Blink LED on PB0 
// Blink LED connected to PB0 — works on ATmega8, ATmega32, ATmega328P
// Compile: avr-gcc -mmcu=atmega8 -Os -o blink.elf blink.c
// Hex:     avr-objcopy -O ihex blink.elf blink.hex

#include <avr/io.h>
#include <util/delay.h>

int main(void) {
    // Set PB0 as output
    DDRB |= (1 << PB0);

    while (1) {
        PORTB |=  (1 << PB0);  // LED ON
        _delay_ms(500);
        PORTB &= ~(1 << PB0);  // LED OFF
        _delay_ms(500);
    }
    return 0;
}
uart.c — ATmega8 · 9600 baud UART transmit 
// Send "Hello ATmega!" over UART at 9600 baud — ATmega8 @ 8MHz
#include <avr/io.h>
#include <util/delay.h>

#define BAUD    9600
#define F_CPU   8000000UL
#define UBRR_V  ((F_CPU / (16UL * BAUD)) - 1)

void uart_init(void) {
    UBRRH = (uint8_t)(UBRR_V >> 8);
    UBRRL = (uint8_t)UBRR_V;
    UCSRB = (1 << TXEN);                  // Enable TX
    UCSRC = (1 << URSEL) | (1 << UCSZ1) | (1 << UCSZ0); // 8-N-1
}

void uart_putc(char c) {
    while (!(UCSRA & (1 << UDRE)));       // Wait TX ready
    UDR = c;
}

void uart_puts(const char *s) {
    while (*s) uart_putc(*s++);
}

int main(void) {
    uart_init();
    while (1) {
        uart_puts("Hello ATmega!\r\n");
        _delay_ms(1000);
    }
}
adc.c — ATmega8/32 · Read analog value from ADC0 
// Read 10-bit ADC value from ADC0 (PC0) — ATmega8/32
#include <avr/io.h>
#include <util/delay.h>

void adc_init(void) {
    // AVcc reference, MUX = ADC0
    ADMUX  = (1 << REFS0);
    // Enable ADC, prescaler = 64 (125kHz @ 8MHz)
    ADCSRA = (1 << ADEN) | (1 << ADPS2) | (1 << ADPS1);
}

uint16_t adc_read(uint8_t channel) {
    ADMUX = (1 << REFS0) | (channel & 0x07);
    ADCSRA |= (1 << ADSC);                // Start conversion
    while (ADCSRA & (1 << ADSC));          // Wait
    return ADC;                             // 0–1023
}

int main(void) {
    adc_init();
    DDRB = 0xFF;                            // PORTB as output
    while (1) {
        uint16_t val = adc_read(0);       // Read ADC0
        // Map 0–1023 to 0–255 and output on PORTB
        PORTB = (uint8_t)(val >> 2);
        _delay_ms(100);
    }
}
pwm.c — ATmega8 · Fast PWM on OC1A (PB1) 
// Hardware Fast PWM on Timer1 OC1A pin — ATmega8
// Connect LED or motor driver to PB1
#include <avr/io.h>
#include <util/delay.h>

void pwm_init(void) {
    DDRB |= (1 << PB1);                    // OC1A as output
    // Fast PWM, 8-bit, non-inverting, prescaler=8
    TCCR1A = (1 << COM1A1) | (1 << WGM10);
    TCCR1B = (1 << WGM12)  | (1 << CS11);
}

void pwm_set(uint8_t duty) {
    OCR1A = duty;                           // 0=off, 255=full
}

int main(void) {
    pwm_init();
    while (1) {
        // Fade up
        for (uint8_t i = 0; i < 255; i++) {
            pwm_set(i);
            _delay_ms(8);
        }
        // Fade down
        for (uint8_t i = 255; i > 0; i--) {
            pwm_set(i);
            _delay_ms(8);
        }
    }
}
timer_isr.c — ATmega8 · 1Hz LED toggle via Timer1 interrupt 
// Toggle PB0 at 1Hz using Timer1 Compare Match interrupt — ATmega8 @ 8MHz
#include <avr/io.h>
#include <avr/interrupt.h>

// Timer1 CTC interrupt — fires every 500ms → 1Hz toggle
ISR(TIMER1_COMPA_vect) {
    PORTB ^= (1 << PB0);                  // Toggle LED
}

int main(void) {
    DDRB |= (1 << PB0);                    // PB0 output

    // Timer1 CTC mode, prescaler=256
    // OCR1A = (8MHz / (256 * 2Hz)) - 1 = 15624
    TCCR1B = (1 << WGM12) | (1 << CS12);
    OCR1A  = 15624;
    TIMSK  = (1 << OCIE1A);               // Enable compare interrupt

    sei();                                  // Global interrupts ON

    while (1) { }                          // Sleep — ISR does the work
}
7seg.c — ATmega8/32 · Count 0–9 on 7-segment display 
// Drive a common-cathode 7-segment display on PORTD — ATmega8/32
// Segments a-g connected to PD0-PD6
#include <avr/io.h>
#include <util/delay.h>

// Segment encoding: bit0=a, bit1=b, ..., bit6=g
const uint8_t seg7[10] = {
    0x3F, /* 0: abcdef  */
    0x06, /* 1: bc       */
    0x5B, /* 2: abdeg    */
    0x4F, /* 3: abcdg    */
    0x66, /* 4: bcfg     */
    0x6D, /* 5: acdfg    */
    0x7D, /* 6: acdefg   */
    0x07, /* 7: abc      */
    0x7F, /* 8: abcdefg  */
    0x6F  /* 9: abcdfg   */
};

int main(void) {
    DDRD = 0x7F;                            // PD0–PD6 as outputs
    while (1) {
        for (uint8_t i = 0; i < 10; i++) {
            PORTD = seg7[i];
            _delay_ms(1000);
        }
    }
}
